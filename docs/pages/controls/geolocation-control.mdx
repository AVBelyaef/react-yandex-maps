# GeolocationControl

Renders Yandex.Maps API [GeolocationControl][control] control

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 9
};

const GeolocationControlExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <GeolocationControl
        options={{ float: 'left' }}
      />
    </Map>
  </YMaps>
)

render(<GeolocationControlExample />);
```
