# Circle

Renders Yandex.Maps API [Circle][]

```jsx
// @live

const mapState = {
  center: [55.76, 37.64],
  zoom: 10
};

const CircleExample = () => (
      <YMaps>
        <Map defaultState={mapState}>
          <Circle
            geometry={[[55.76, 37.60], 10000]}
            options={{
              draggable: true,
              fillColor: "#DB709377",
              strokeColor: "#990066",
              strokeOpacity: 0.8,
              strokeWidth: 5
            }}
          />
        </Map>
      </YMaps>
    );

render(<CircleExample />);
```
