# FullscreenControl

Renders Yandex.Maps API [FullscreenControl][control] control

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 3
};

const FullscreenControlExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <FullscreenControl />
    </Map>
  </YMaps>
)

render(<FullscreenControlExample />);
```
