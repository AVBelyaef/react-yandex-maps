# ListBox and ListBoxItem

Renders Yandex.Maps API [ListBox][listbox] and [ListBoxItem][listboxitem] controls

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 9
};

const ListBoxExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <ListBox data={{ content: 'Select city' }}>
        <ListBoxItem data={{ content: 'Moscow' }} />
        <ListBoxItem data={{
          content: 'Saint Petersburg'
        }} />
      </ListBox>
    </Map>
  </YMaps>
)

render(<ListBoxExample />);
```
