# Polygon

Renders Yandex.Maps API [Polygon][]

```jsx
// @live

const mapState = {
  center: [55.73, 37.90],
  zoom: 10
};

const PolygonExample = () => (
      <YMaps>
        <Map defaultState={mapState}>
          <Polygon
            geometry={[
              [[55.75, 37.8], [55.8, 37.9], [55.75, 38.0], [55.7, 38.0], [55.7, 37.8]],
              [[55.75, 37.82], [55.75, 37.98], [55.65, 37.9]]
            ]}
            options={{
              fillColor: '#00FF00',
              strokeColor: '#0000FF',
              opacity: 0.5,
              strokeWidth: 5,
              strokeStyle: 'shortdash'
            }}
          />
        </Map>
      </YMaps>
    );

render(<PolygonExample />);
```
