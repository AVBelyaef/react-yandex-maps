# Button

Renders Yandex.Maps API [Button][control] control

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 3,
};

const ButtonExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <Button
        data={{ content: 'Unpress me!' }}
        options={{ maxWidth: 128 }}
        defaultState={{ selected: true }}
      />
    </Map>
  </YMaps>
);

render(<ButtonExample />);
```
