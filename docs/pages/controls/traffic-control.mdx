# TrafficControl

Renders Yandex.Maps API [TrafficControl][control] control

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 9,
};

const TrafficControlExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <TrafficControl
        defaultState={{
          trafficShown: true,
          providerKey: 'traffic#archive',
        }}
      />
    </Map>
  </YMaps>
);

render(<TrafficControlExample />);
```
