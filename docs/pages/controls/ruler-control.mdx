# RulerControl

Renders Yandex.Maps API [RulerControl][control] control

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 3,
};

const RulerControlExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <RulerControl options={{ scaleLine: false }} />
    </Map>
  </YMaps>
);

render(<RulerControlExample />);
```
