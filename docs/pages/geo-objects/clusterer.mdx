# Clusterer

Renders Yandex.Maps API [GeoObject][]

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 5,
};

const ClustererExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <Clusterer
        options={{
          preset: 'islands#invertedVioletClusterIcons',
          groupByCoordinates: false,
        }}
      >
        {points.map((coordinates, index) => (
          <Placemark key={index} geometry={coordinates} />
        ))}
      </Clusterer>
    </Map>
  </YMaps>
);

render(<ClustererExample />);
```
