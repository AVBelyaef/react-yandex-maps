# TypeSelector

Renders Yandex.Maps API [TypeSelector][control] control

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 9,
};

const TypeSelectorExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <TypeSelector defaultState={{ expanded: true }} />
    </Map>
  </YMaps>
);

render(<TypeSelectorExample />);
```
