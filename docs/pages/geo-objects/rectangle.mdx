# Rectangle

Renders Yandex.Maps API [Rectangle][]

```jsx
// @live

const mapState = {
  center: [55.674, 37.601],
  zoom: 11,
};

const RectangleExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <Rectangle
        geometry={[[55.66, 37.6], [55.71, 37.69]]}
        options={{
          draggable: true,
          fillColor: '#ffff0022',
          strokeColor: '#3caa3c88',
          strokeWidth: 7,
        }}
      />
    </Map>
  </YMaps>
);

render(<RectangleExample />);
```
