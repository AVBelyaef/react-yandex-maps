# RouteButton

Renders Yandex.Maps API [RouteButton][control] control

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 9
};

const RouteButtonExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <RouteButton options={{ float: 'right' }} />
    </Map>
  </YMaps>
)

render(<RouteButtonExample />);
```
