# GeoObject

Renders Yandex.Maps API [GeoObject][]

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 3,
};

const GeoObjectExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <GeoObject
        geometry={{
          type: 'Point',
          coordinates: mapState.center,
        }}
      />
      <GeoObject
        geometry={{
          type: 'LineString',
          coordinates: [mapState.center, [52.51, 13.38]],
        }}
        options={{
          geodesic: true,
          strokeWidth: 5,
          strokeColor: '#F008',
        }}
      />
    </Map>
  </YMaps>
);

render(<GeoObjectExample />);
```
