# RouteEditor

Renders Yandex.Maps API [RouteEditor][control] control

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 9,
};

const RouteEditorExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <RouteEditor
        data={{ title: 'Click this to create a route' }}
        options={{ float: 'right' }}
      />
    </Map>
  </YMaps>
);

render(<RouteEditorExample />);
```
