# Polyline

Renders Yandex.Maps API [Polyline][]

```jsx
// @live

const mapState = {
  center: [55.72, 37.44],
  zoom: 10
};

const PolylineExample = () => (
      <YMaps>
        <Map defaultState={mapState}>
          <Polyline
            geometry={[
              [55.8, 37.5],
              [55.8, 37.4],
              [55.7, 37.5],
              [55.7, 37.4],
            ]}
            options={{
              balloonCloseButton: false,
              strokeColor: "#000",
              strokeWidth: 4,
              strokeOpacity: 0.5
            }}
          />
        </Map>
      </YMaps>
    );

render(<PolylineExample />);
```
