# SearchControl

Renders Yandex.Maps API [SearchControl][control] control

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 3,
};

const SearchControlExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <SearchControl
        options={{
          noPlacemark: true,
          placeholderContent: 'This is search control',
        }}
      />
    </Map>
  </YMaps>
);

render(<SearchControlExample />);
```
