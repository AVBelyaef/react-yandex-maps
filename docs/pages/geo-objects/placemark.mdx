# Placemark

Renders Yandex.Maps API [Placemark][]

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 5,
};

const PlacemarkExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <Placemark geometry={[55.684758, 37.738521]} />
    </Map>
  </YMaps>
);

render(<PlacemarkExample />);
```
