# ZoomControl

Renders Yandex.Maps API [ZoomControl][control] control

```jsx
// @live

const mapState = {
  center: [55.751574, 37.573856],
  zoom: 3
};

const ZoomControlExample = () => (
  <YMaps>
    <Map defaultState={mapState}>
      <ZoomControl 
        options={{ 
          size: 'small',
          zoomDuration: 1000,
        }}
      />
    </Map>
  </YMaps>
)

render(<ZoomControlExample />);
```
